//一个前缀积，一个后缀积
```c
class Solution {
public:
    vector<int> multiply(const vector<int>& A) {
        if(A.empty()) return{};
        int n=A.size(),j=1;
        vector<int> ans(n,1);
        ans[n-1]=A[n-1];
        for(int i=n-2;i>=0;i--) ans[i]=ans[i+1]*A[i];
        for(int i=0;i<n-1;i++){
            ans[i]=j*ans[i+1];
            j*=A[i];
        }
        ans[n-1]=j;
        return ans;
    }
};
```
