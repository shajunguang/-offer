//二叉搜索树的中序遍历，便是有序的，递归思路，先转换左边的，把左边最后指针和root接上，再转换右边的，把root和首指针接上即可
```c
/*
struct TreeNode {
	int val;
	struct TreeNode *left;
	struct TreeNode *right;
	TreeNode(int x) :
			val(x), left(NULL), right(NULL) {
	}
};*/
class Solution {
public:
    TreeNode* Convert(TreeNode* root)
    {
        if(root==NULL)return root;
        if(root->left==NULL && root->right==NULL) return root;
        auto lt=Convert(root->left);
        auto p=lt;
        while(p!=NULL && p->right!=NULL) p=p->right;
        if(p!=NULL){
            p->right=root;
            root->left=p;
        }
        auto rt=Convert(root->right);
        if(rt!=NULL){
            rt->left=root;
            root->right=rt;
        }
        return lt==NULL?root:lt; //lt还要用，因此前面修改指针的时候，要用替代指针修改，精髓了！
    }
};
```
//迭代版本,利用栈模拟递归过程，给首指针打上标记
```c
/*
struct TreeNode {
	int val;
	struct TreeNode *left;
	struct TreeNode *right;
	TreeNode(int x) :
			val(x), left(NULL), right(NULL) {
	}
};*/
class Solution {
public:
    TreeNode* Convert(TreeNode* root)
    {
        if(!root) return root;
        stack<TreeNode*> st;
        auto p=root;
        TreeNode* pre=NULL;//保存前面节点
        bool isfirst=true;
       while(p!=NULL || !st.empty()){
           while(p){
               st.push(p);
               p=p->left;
           }
           p=st.top();st.pop();
           if(isfirst){
               root=p;
               isfirst=false;
               pre=root;
           }else{
               pre->right=p;
               p->left=pre;
               pre=p;
           }
           p=p->right;
       }
        return root;
    }
};
```
